export const Template = (args, { argTypes, updateArgs }) => ({
    props: Object.keys(argTypes),
    components: { GRadio },
    template: `
        <g-radio v-bind="filteredProps" @input="handleInput">
            {{slotDefault}}
        </g-radio>
    `,
    computed: {
        filteredProps() {
            const { slotDefault = null, ...props } = this.$props
            return props
        },
    },
    methods: {
        handleInput(value) {
            updateArgs({ ...args, value })
        },
    },
})
